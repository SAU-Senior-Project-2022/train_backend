Documentation

-- GET:

ENDPOINT: /state/[station_id]
PURPOSE: to get the state of a specific sensor station
RETURN: {"time": dict, "error_message": str, "error_state": bool, "id": int "state": int, "station_id": int}

ENDPOINT: /location
PURPOSE: to get the location info of ALL sensor stations
RETURN: {["error_message": str, "error_state": bool, "id": int, "latitude": float, "longitude": float]}

ENDPOINT: /location/[station_id]
PURPOSE: to get the location info of a specific sensor station
RETURN: {"error_message": str, "error_state": bool, "id": int, "latitude": float, "longitude":float}

-- POST:

-- Updating a location state

ENDPOINT: /state/[station_id]
PURPOSE: to update the state of a specific sensor station
BODY: {"state": bool}
RETURN: {"success": bool}

-- Adding a new location:

ENDPOINT: /location/new
PURPOSE: to register a new sensor station and give it an ID
BODY: {"latitude": bool, "longitude": bool, "title": string}
RETURN: {"station_id": int}
